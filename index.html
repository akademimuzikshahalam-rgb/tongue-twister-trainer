  <script>

    const tongueTwisters = [

      "She sells seashells by the seashore",

      "Peter Piper picked a peck of pickled peppers",

      "Red lorry yellow lorry",

      "Unique New York unique New York",

      "How much wood would a woodchuck chuck"

    ];


    let currentIndex = 0;

    let recognition = null;

    let isRecording = false;

    let transcript = "";


    const twisterText = document.getElementById("tongue-twister");

    const numberText = document.getElementById("twister-number");

    const totalText = document.getElementById("total-twisters");

    const recordBtn = document.getElementById("record-btn");

    const statusText = document.getElementById("record-status");

    const resultBox = document.getElementById("results");

    const transcriptBox = document.getElementById("transcript");

    const scoreRing = document.getElementById("score-ring");

    const scoreValue = document.getElementById("score-value");

    const feedbackBox = document.getElementById("feedback");


    totalText.innerText = tongueTwisters.length;


    function updateTwister() {

      twisterText.innerText = tongueTwisters[currentIndex];

      numberText.innerText = currentIndex + 1;

    }


    updateTwister();


    function initRecognition() {

      const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;


      if (!SpeechRecognition) {

        statusText.innerText = "⚠️ Speech recognition not supported. Use Chrome, Edge, or Safari.";

        return false;

      }


      recognition = new SpeechRecognition();

      recognition.continuous = false;

      recognition.interimResults = true;

      recognition.lang = "en-US";


      recognition.onresult = (e) => {

        transcript = e.results[e.results.length - 1][0].transcript;

        showResult();

      };


      recognition.onerror = (e) => {

        console.error("Speech Recognition Error Details:", {

          error: e.error,

          timestamp: new Date().toISOString(),

          userAgent: navigator.userAgent,

          protocol: window.location.protocol

        });

        

        let errorMsg = `❌ Error: ${e.error}`;

        if (e.error === "not-allowed") {

          errorMsg = "❌ Permission denied. Microphone access blocked.";

        } else if (e.error === "no-speech") {

          errorMsg = "❌ No speech detected. Try speaking louder.";

        } else if (e.error === "network") {

          errorMsg = "❌ Network error. Check your connection.";

        } else if (e.error === "service-not-allowed") {

          errorMsg = "❌ Service not allowed in this context.";

        }

        

        statusText.innerText = errorMsg;

        stopRecording();

      };


      return true;

    }


    function startRecording() {

      if (!recognition) {

        initRecognition().then(ok => {

          if (ok) startRecording();

        });

        return;

      }


      isRecording = true;

      recordBtn.classList.add("listening");

      statusText.innerText = "Listening...";

      recognition.start();

    }


    function stopRecording() {

      isRecording = false;

      recordBtn.classList.remove("listening");

      recognition.stop();

    }


    recordBtn.addEventListener("click", () => {

      if (!recognition) {

        const ok = initRecognition();

        if (!ok) {

          return;

        }

      }


      if (isRecording) {

        stopRecording();

      } else {

        startRecording();

      }

    });


    function calculateScore(original, spoken) {

      const o = original.toLowerCase().split(" ");

      const s = spoken.toLowerCase().split(" ");


      if (s.length < o.length * 0.6) {

        return 1;

      }


      let match = 0;

      o.forEach(word => {

        if (s.includes(word)) match++;

      });


      let accuracy = match / o.length;

      let score = Math.round(accuracy * 10);


      if (score < 1) score = 1;

      if (score > 10) score = 10;


      return score;

    }


    function showResult() {

      const original = tongueTwisters[currentIndex];

      const score = calculateScore(original, transcript);


      transcriptBox.innerText = transcript;

      scoreValue.innerText = score;

      scoreRing.style.setProperty("--score-percent", (score * 10) + "%");


      let feedback = "";

      if (score >= 9) feedback = "🌟 Perfect pronunciation!";

      else if (score >= 7) feedback = "👏 Very clear speech!";

      else if (score >= 5) feedback = "👍 Good effort. Try again!";

      else feedback = "💪 Slow down and pronounce each word clearly.";


      feedbackBox.innerText = feedback;

      resultBox.classList.add("show");

      statusText.innerText = "Tap to record again";

    }


    document.getElementById("try-again").addEventListener("click", () => {

      resultBox.classList.remove("show");

      statusText.innerText = "Tap to start recording";

    });


    document.getElementById("next-btn").addEventListener("click", () => {

      currentIndex = (currentIndex + 1) % tongueTwisters.length;

      updateTwister();

      resultBox.classList.remove("show");

      statusText.innerText = "Tap to start recording";

    });

  </script>
